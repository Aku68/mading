        document.addEventListener('DOMContentLoaded', function() {

            // Fungsi untuk jam dan tanggal digital

            function updateClock() {

                const now = new Date();

                const jam = String(now.getHours()).padStart(2, '0');

                const menit = String(now.getMinutes()).padStart(2, '0');

                const detik = String(now.getSeconds()).padStart(2, '0');

                const jamDigitalEl = document.getElementById('jam-digital');

                if (jamDigitalEl) {

                    jamDigitalEl.textContent = `${jam}:${menit}:${detik}`;

                }


                const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric', timeZone: 'Asia/Jakarta' };

                const tanggalDigitalEl = document.getElementById('tanggal-digital');

                if(tanggalDigitalEl) {

                    tanggalDigitalEl.textContent = now.toLocaleDateString('id-ID', options);

                }

            }


            // Panggil updateClock setiap detik

            setInterval(updateClock, 1000);

            updateClock(); // Panggil sekali saat halaman dimuat


            // Fungsi untuk navigasi tab

            const tabButtons = document.querySelectorAll('.tab-button');

            const tabContents = document.querySelectorAll('.tab-content');


            tabButtons.forEach(button => {

                button.addEventListener('click', () => {

                    const tabId = button.getAttribute('data-tab');


                    // Atur tombol aktif

                    tabButtons.forEach(btn => btn.classList.remove('active'));

                    button.classList.add('active');


                    // Tampilkan konten yang sesuai

                    tabContents.forEach(content => {

                        if (content.id === `${tabId}-content`) {

                            content.classList.add('active', 'fade-in');

                        } else {

                            content.classList.remove('active', 'fade-in');

                        }

                    });

                });

            });

        });
